\section*{Examples}

\begin{itemize}

\item English Intervocalic Alveolar Flapping:

\begin{verbatim}
  
  \begin{phon_rule}
  \cvfeatures{C}{\feature{+}{stop} \feature{+}{alv.}} \> \phone{R} \/ 
  \cvfeatures{V}{\feature{+}{stress}} \_ \cvfeatures{V}{\feature{-}{stress}}
  \end{phon_rule}
    
\end{verbatim}

\begin{phon_rule}
\cvfeatures{C}{\feature{+}{stop} \feature{+}{alv.}} \> \phone{R} \/ 
\cvfeatures{V}{\feature{+}{stress}} \_ \cvfeatures{V}{\feature{-}{stress}}
\end{phon_rule}

\item English Vowel Nasalization:

\begin{verbatim}
  
  \begin{phon_rule}
  \cv{V} \> \features{\feature{+}{nasal}} \/ 
  \_ \features{\feature{+}{nasal}} \cv{(C)} \#
  \end{phon_rule}
    
\end{verbatim}

\begin{phon_rule}
\cv{V} \> \features{\feature{+}{nasal}} \/ 
\_ \features{\feature{+}{nasal}} \cv{(C)} \#
\end{phon_rule}

\item Yoruba Tone Falling:

\begin{verbatim}
  \cv{L} \> \cv{F} \/
  \cv{H} \_
\end{verbatim}

\begin{phon_rule}
\cv{L} \> \cv{F} \/
\cv{H} \_
\end{phon_rule}

\item English \ipa{@} Insertion:

\begin{verbatim}
  \begin{phon_rule}
  \null \> \phone{@} \/
  \features{\feature{+}{sib.}} \_ \features{\feature{+}{sib.}} \#
  \end{phon_rule}
\end{verbatim}

\begin{phon_rule}
\null \> \phone{@} \/
\features{\feature{+}{sib.}} \_ \features{\feature{+}{sib.}} \#
\end{phon_rule}

\item English Nasal Place Assimilation:

\begin{verbatim}
  \begin{phon_rule}
  \cv{N} \> \features{\feature{\alpha}{place}} \/
  \_ \cvfeatures{C}{\feature{\alpha}{place}}
  \end{phon_rule}
\end{verbatim}

\begin{phon_rule}
\cv{N} \> \features{\feature{\alpha}{place}} \/
\_ \cvfeatures{C}{\feature{\alpha}{place}}
\end{phon_rule}

\item Turkish Sonorant Devoicing:

\begin{verbatim}
  \begin{phon_rule}
  \features{\feature{-}{sonorant}} \> \features{\feature{-}{voice}} \/
  \_ \oneof{\# \\ \cv{C}}
  \end{phon_rule}
\end{verbatim}

\begin{phon_rule}
\features{\feature{-}{sonorant}} \> \features{\feature{-}{voice}} \/
\_ \oneof{\# \\ \cv{C}}
\end{phon_rule}

\end{itemize}
